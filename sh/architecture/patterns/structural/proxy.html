Реальный объект и прокси имплементируют один интерфейс
Прокси замещает собой реальный объект.
Прокси содержит в себе экземпляр объекта и в своих методах вызывает его методы.
+ доп функциональность (проверка прав, логирование ...)
    // Интерфейс
    interface Image {
        void display();
    }

    // Реальный объект
    class RealImage implements Image {
        @Override
        public void display() {
            //показывает Image
        }
    }

    // Прокси
    class ProxyImage implements Image {
        @Override
        public void display() {
            if (realImage == null) {
                realImage = new RealImage(filename);
            }
            realImage.display();
            log.debug("Display Image!"); // ДОП ФУНКЦИОНАЛЬНОСТЬ
        }
    }

    // Клиентский код
    public class ProxyPatternDemo {
        public static void main(String[] args) {
            Image image = new ProxyImage();
            image.display(); 
        }
    }
