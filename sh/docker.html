docker version : показать версию

docker ps -a : показать контейнеры
-a и остановленные

docker images : показать локальные образы

docker run -it -d -p 8080:80 -v ${PWD}:/usr/share/nginx/htmll --name --rm my_nginx nginx
: создаёт (скачивает образ если нет) и запускает контейнер
-it: интерактивный терминал
-d: в фоновом режимие detached
-p 8080:80 : публиковать порт внешний:контейнера
-v ${PWD}:/usr/share/nginx/htmll : подключить том. локальная_папка:папка_контейнера
--name my_nginx : задать имя
--rm : удалить как остановится
nginx: название образа

docker exec cont_name bash : выполнить команду в запущенном контейнере
cont_name : имя или id контейнера
bash : название процесса

docker container prune : удалить все остановленные контейнеры

docker container inspect cont_name | grep IPAddress : выдать информацию о контейнере
cont_name : имя или id контейнера
grep : фильтровать по полю

docker stop cont_name cont_name2 : остановить контейнер
cont_name : имя или id контейнера

docker build . -t my-calendar:4.1.3 : создать образ
. : путь к Dockerfile
-t my-calendar:4.1.3 : добавить имя:тэг(версия)

\ : перенос строки

Dockerfile :
FROM python:alpine имя базового образа : тэг(версия) базового образа
WORKDIR /app
COPY . . локаольная папка, папка в контейнере
CMD ["python", "main.py"]

docker-compose.yml : Декларативный. Пробелы важны
version: '3'
services:
  app:
    build: ./app
  mongo:
    image: mongo
